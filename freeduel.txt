local UserInputService = game:GetService("UserInputService")
local function makeDraggable(handle, target)
    local dragging = false
    local dragStart
    local startPosition
    local dragInput

    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPosition = target.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then dragging = false end
            end)
        end
    end)

    handle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and input == dragInput then
            local delta = input.Position - dragStart
            target.Position = UDim2.new(startPosition.X.Scale, startPosition.X.Offset + delta.X, startPosition.Y.Scale, startPosition.Y.Offset + delta.Y)
        end
    end)
end

local function novuSerialize(value, depth)
    depth = depth or 0
    if depth > 6 then return "nil" end

    local valueType = typeof(value)
    if valueType == "boolean" or valueType == "number" then
        return tostring(value)
    elseif valueType == "string" then
        return string.format("%q", value)
    elseif valueType == "table" then
        local parts = {"{"}
        for key, item in pairs(value) do
            local encodedKey
            if type(key) == "string" and key:match("^[%a_][%w_]*$") then
                encodedKey = key
            else
                encodedKey = "[" .. novuSerialize(key, depth + 1) .. "]"
            end
            table.insert(parts, encodedKey .. "=" .. novuSerialize(item, depth + 1) .. ",")
        end
        table.insert(parts, "}")
        return table.concat(parts)
    end
    return "nil"
end

local function novuSaveSettings()
    -- getgenv yedeği, dosya desteği olmayan executorlerde de aynı oturumda çalışır.
    getgenv().NovuHubSavedSettings = novuSaved

    if type(writefile) ~= "function" then
        return false
    end

    local okEncode, payload = pcall(function()
        return HttpService:JSONEncode(novuSaved)
    end)
    if not okEncode then return false end

    local saved = false
    local okRoot = pcall(function()
        writefile(NOVU_SAVE_FILE, payload)
    end)
    saved = okRoot or saved

    if type(makefolder) == "function" then
        pcall(function()
            if type(isfolder) ~= "function" or not isfolder("NovuHub") then
                makefolder("NovuHub")
            end
        end)
        local okFolder = pcall(function()
            writefile(NOVU_SAVE_FILE_FOLDER, payload)
        end)
        saved = okFolder or saved
    end

    return saved
end

local function novuApplyLoadedTable(loaded)
    if type(loaded) ~= "table" then return false end
    for key, value in pairs(loaded) do
        if key == "Keys" and type(value) == "table" then
            novuSaved.Keys = novuSaved.Keys or {}
            for keyName, keyValue in pairs(value) do
                novuSaved.Keys[keyName] = keyValue
            end
        else
            novuSaved[key] = value
        end
    end
    return true
end

local function novuLoadJsonFile(path)
    if type(isfile) ~= "function" or type(readfile) ~= "function" then return false end
    local exists = false
    pcall(function() exists = isfile(path) end)
    if not exists then return false end

    local ok, loaded = pcall(function()
        return HttpService:JSONDecode(readfile(path))
    end)
    return ok and novuApplyLoadedTable(loaded)
end

local function novuLoadLegacyFile(path)
    if type(isfile) ~= "function" or type(readfile) ~= "function" then return false end
    local exists = false
    pcall(function() exists = isfile(path) end)
    if not exists then return false end
    local loader = loadstring or load
    if not loader then return false end
    local ok, loaded = pcall(function()
        local chunk = loader(readfile(path))
        return chunk and chunk()
    end)
    return ok and novuApplyLoadedTable(loaded)
end

    if type(getgenv().NovuHubSavedSettings) == "table" then
        novuApplyLoadedTable(getgenv().NovuHubSavedSettings)
    end
end

            root.Velocity = Vector3.new(
                lastMoveDirection.X * activeSpeed,
                root.Velocity.Y,
                lastMoveDirection.Z * activeSpeed
            )
        end
    end
end

        elseif object:IsA("ParticleEmitter")
            or object:IsA("Trail")

            or object:IsA("Fire")
            or object:IsA("Smoke")

-- ANTI RADGOLL.txt sistemi, GUI düğmesine bağlı ve yeniden doğmada tekrar hazırlanır.
local antiRagdollEnabled = false
local antiRagdollRenderConn = nil
local antiRagdollCharacterConn = nil
local ragdollCache = {}

local function cacheRagdollCharacter(character)
    character = character or player.Character
    if not character then return false end

    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local root = character:FindFirstChild("HumanoidRootPart")
    if not humanoid or not root then return false end

    ragdollCache = {
        character = character,
        humanoid = humanoid,
        root = root,
    }

    local camera = workspace.CurrentCamera
    if camera then
        camera.CameraSubject = humanoid
    end
    return true
end

local function isCurrentlyRagdolled()
    local humanoid = ragdollCache.humanoid
    if not humanoid then return false end

    local state = humanoid:GetState()
    if state == Enum.HumanoidStateType.Physics
        or state == Enum.HumanoidStateType.Ragdoll
        or state == Enum.HumanoidStateType.FallingDown then
        return true
    end

    local endTime = player:GetAttribute("RagdollEndTime")
    if endTime then
        local now = workspace:GetServerTimeNow()
        if (endTime - now) > 0 then
            return true
        end
    end

    return false
end

local function removeRagdollConstraints()
    local character = ragdollCache.character
    if not character then return end

    for _, object in ipairs(character:GetDescendants()) do
        if object:IsA("BallSocketConstraint")
            or (object:IsA("Attachment") and object.Name:find("RagdollAttachment")) then
            pcall(function()
                object:Destroy()
            end)
        end
    end
end

local function forceExitRagdoll()
    local humanoid = ragdollCache.humanoid
    local root = ragdollCache.root
    if not humanoid or not root then return end

    pcall(function()
        player:SetAttribute("RagdollEndTime", workspace:GetServerTimeNow())
    end)

    if humanoid.Health > 0 then
        humanoid:ChangeState(Enum.HumanoidStateType.Running)
    end

    root.Anchored = false
    root.AssemblyLinearVelocity = Vector3.zero
    root.AssemblyAngularVelocity = Vector3.zero
    root.CFrame = CFrame.new(root.Position) * CFrame.Angles(0, math.rad(root.Orientation.Y), 0)

    local camera = workspace.CurrentCamera
    if camera and camera.CameraSubject ~= humanoid then
        camera.CameraSubject = humanoid
    end
end

local function stopAntiRagdoll()
    antiRagdollEnabled = false

    if antiRagdollRenderConn then
        antiRagdollRenderConn:Disconnect()
        antiRagdollRenderConn = nil
    end
    if antiRagdollCharacterConn then
        antiRagdollCharacterConn:Disconnect()
        antiRagdollCharacterConn = nil
    end
    table.clear(ragdollCache)
end

local function startAntiRagdoll()
    stopAntiRagdoll()
    antiRagdollEnabled = true
    cacheRagdollCharacter(player.Character)

    antiRagdollRenderConn = RunService.RenderStepped:Connect(function()
        if not antiRagdollEnabled then return end

        if not ragdollCache.humanoid or not ragdollCache.humanoid.Parent then
            cacheRagdollCharacter(player.Character)
            return
        end

        if isCurrentlyRagdolled() then
            removeRagdollConstraints()
            forceExitRagdoll()
        end
    end)

local function useMedusaCounter()
    if medusaDebounce then return end
    if MEDUSA_COOLDOWN > (tick() - medusaLastUsed) then return end

    local character = player.Character
    if not character then return end

    medusaDebounce = true
    local medusa = findMedusaCounterTool()
    if not medusa then
        medusaDebounce = false
        return
    end

    if medusa.Parent ~= character then
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            pcall(function()
                humanoid:EquipTool(medusa)
            end)
        end
    end

    pcall(function()
        medusa:Activate()
    end)

    medusaLastUsed = tick()
    medusaDebounce = false

local function disconnectMedusaCounter()
    for _, connection in ipairs(medusaAnchorConnections) do
        pcall(function()
            connection:Disconnect()
        end)
    end
    table.clear(medusaAnchorConnections)
end

local function watchMedusaPart(part)
    local connection = part:GetPropertyChangedSignal("Anchored"):Connect(function()
        if medusaCounterEnabled and part.Anchored and part.Transparency == 1 then
            useMedusaCounter()
        end
    end)
    table.insert(medusaAnchorConnections, connection)
end

    disconnectMedusaCounter()
    if not character then return end

    for _, part in ipairs(character:GetDescendants()) do
        if part:IsA("BasePart") then

    for _, plot in ipairs(plots:GetChildren()) do

        local pods = plot:FindFirstChild("AnimalPodiums")
        if not pods then continue end

                            and p.ActionText
                            and p.ActionText:find("Steal") then

            end

            for _, c in ipairs(getconnections(prompt.PromptButtonHoldBegan)) do

                end
            end

            for _, c in ipairs(getconnections(prompt.Triggered)) do
                if c.Function then
                    table.insert(StealData[prompt].trigger, c.Function)
                end
            end
        end
    end

    local data = StealData[prompt]
    if not data.ready then return end

    data.ready = false
    isStealing = true
    local startTime = tick()

    task.spawn(function()
        for _, f in ipairs(data.hold) do

            local elapsed = tick() - startTime
            local p = math.clamp(elapsed / STEAL_DURATION, 0, 1)
            updateProgressBar(p)
            task.wait()
        end

        updateProgressBar(1)

        for _, f in ipairs(data.trigger) do
            pcall(f)
        end

        task.wait(0.05)
        updateProgressBar(0)
        data.ready = true
        isStealing = false

local function fadePanels(alpha, duration)
    for _, object in ipairs(guiObjects) do
        if object.Parent then
            local original = originalTransparency[object]
            local goals = {}
            if original.BackgroundTransparency ~= nil then
                goals.BackgroundTransparency = original.BackgroundTransparency + (1 - original.BackgroundTransparency) * alpha
            end
            if original.TextTransparency ~= nil then
                goals.TextTransparency = original.TextTransparency + (1 - original.TextTransparency) * alpha
            end
            if original.TextStrokeTransparency ~= nil then
                goals.TextStrokeTransparency = original.TextStrokeTransparency + (1 - original.TextStrokeTransparency) * alpha
            end
            if original.ImageTransparency ~= nil then
                goals.ImageTransparency = original.ImageTransparency + (1 - original.ImageTransparency) * alpha
            end
            if next(goals) then
                TweenService:Create(object, TweenInfo.new(duration, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), goals):Play()
            end
        end
    end
end

-- ─── Helper Functions ──────────────────────────────────────────────────────

local function findAnyTool()
    local c = LP.Character
    if c then 
        for _, v in ipairs(c:GetChildren()) do 
            if v:IsA("Tool") then return v end 
        end 
    end
    local bp = LP:FindFirstChildOfClass("Backpack")
    if bp then 
        for _, v in ipairs(bp:GetChildren()) do 
            if v:IsA("Tool") then return v end 
        end 
    end
    return nil
end

local function findBat()
    local char = LP.Character
    if not char then return nil end
    
    -- Check equipped tools
    for _, tool in ipairs(char:GetChildren()) do
        if tool:IsA("Tool") then
            local name = tool.Name:lower()
            if name:find("bat") or name:find("slap") then 
                return tool 
            end
        end
    end
    
    -- Check backpack
    local bp = LP:FindFirstChild("Backpack")
    if bp then
        for _, tool in ipairs(bp:GetChildren()) do
            if tool:IsA("Tool") then
                local name = tool.Name:lower()
                if name:find("bat") or name:find("slap") then 
                    return tool 
                end
            end
        end
    end
    
    return nil
end

local function getClosestTarget()
    local root = LP.Character and LP.Character:FindFirstChild("HumanoidRootPart")
    if not root then return nil end
    
    local closest, minDist = nil, math.huge
    
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= LP and plr.Character then
            local tRoot = plr.Character:FindFirstChild("HumanoidRootPart")
            local hum = plr.Character:FindFirstChildOfClass("Humanoid")
            
            if tRoot and hum and hum.Health > 0 then
                local dist = (tRoot.Position - root.Position).Magnitude
                if dist < minDist then 
                    minDist = dist
                    closest = tRoot 
                end
            end
        end
    end
    
    return closest
end

local function tryHitBat()
    if State.hittingCooldown then return end
    State.hittingCooldown = true
    
    pcall(function()
        local c = LP.Character
        if not c then return end
        
        local hum = c:FindFirstChildOfClass("Humanoid")
        local tool = findAnyTool()
        
        if tool then
            -- Equip if not already equipped
            if tool.Parent ~= c and hum then 
                pcall(function() hum:EquipTool(tool) end) 
            end
            
            -- Try to swing
            local remote = tool:FindFirstChildOfClass("RemoteEvent")
            if remote then 
                pcall(function() remote:FireServer() end)
            else 
                pcall(function() tool:Activate() end) 
            end
        end
    end)
    
    task.delay(MOB_SWING_COOLDOWN, function() 
        State.hittingCooldown = false 
    end)
end

local function startBatAimbot()
    -- Disconnect existing aimbot
    if Conns.aimbot then 
        Conns.aimbot:Disconnect() 
        Conns.aimbot = nil
    end
    
    -- Disable conflicting features
    if State.autoLeftEnabled then 
        State.autoLeftEnabled = false
        stopAutoLeft() 
    end
    if State.autoRightEnabled then 
        State.autoRightEnabled = false
        stopAutoRight() 
    end

    -- Disable auto-rotation
    local hum = LP.Character and LP.Character:FindFirstChildOfClass("Humanoid")
    if hum then 
        hum.AutoRotate = false 
    end

    -- Start aimbot loop
    Conns.aimbot = RunService.RenderStepped:Connect(function()
        if not State.autoBatToggled then return end
        
        local char = LP.Character
        if not char then return end
        
        local root = char:FindFirstChild("HumanoidRootPart")
        if not root then return end
        
        local hum = char:FindFirstChildOfClass("Humanoid")
        if not hum then return end

        -- Auto-equip bat if not holding any tool
        if not char:FindFirstChildOfClass("Tool") then
            local bat = findBat()
            if bat then 
                pcall(function() hum:EquipTool(bat) end) 
            end
        end

        -- Get closest target
        local target = getClosestTarget()
        if not target then return end

        -- Target prediction
        local targetVel = target.AssemblyLinearVelocity
        local myPos = root.Position
        local targetPos = target.Position

        -- Predict position (0.14 seconds ahead + look vector offset)
        local predictPos = targetPos + targetVel * 0.14
        predictPos = predictPos + target.CFrame.LookVector * 0.3

        -- Calculate movement direction
        local direction = predictPos - myPos
        local flatDir = Vector3.new(direction.X, 0, direction.Z).Unit
        local chaseSpeed = 58

        -- Vertical movement with height advantage
        local desiredHeight = targetPos.Y + 3.7
        local yVel = (desiredHeight - myPos.Y) * 19.5 + targetVel.Y * 0.8
        
        if hum.FloorMaterial ~= Enum.Material.Air then
            yVel = math.max(yVel, 13)
        end
        yVel = math.clamp(yVel, -70, 110)

        -- Apply velocity with smoothing
        local desiredVel = Vector3.new(
            flatDir.X * chaseSpeed, 
            yVel, 
            flatDir.Z * chaseSpeed
        )
        root.AssemblyLinearVelocity = root.AssemblyLinearVelocity:Lerp(desiredVel, 0.8)

        -- 3D Look at target with tilt (Dark Hub style)
        local speed3 = targetVel.Magnitude
        local predictTime = math.clamp(speed3 / 150, 0.05, 0.2)
        local predictedPos = targetPos + targetVel * predictTime
        local toPredict = predictedPos - myPos
        
        if toPredict.Magnitude > 0.1 then
            local goalCF = CFrame.lookAt(myPos, predictedPos)
            local curCF = root.CFrame
            local diffCF = curCF:Inverse() * goalCF
            local rx, ry, rz = diffCF:ToEulerAnglesXYZ()
            
            -- Clamp rotations to prevent spinning
            rx = math.clamp(rx, -2.5, 2.5)
            ry = math.clamp(ry, -2.5, 2.5)
            rz = math.clamp(rz, -2.5, 2.5)
            
            local tiltSpeed = 42
            root.AssemblyAngularVelocity = root.CFrame:VectorToWorldSpace(
                Vector3.new(rx * tiltSpeed, ry * tiltSpeed, rz * tiltSpeed)
            )
        end
        
        -- Auto-swing at target (optional)
        -- Uncomment to enable automatic hitting
        -- if (target.Position - myPos).Magnitude < 10 then
        --     tryHitBat()
        -- end
    end)
end

    local hum = c and c:FindFirstChildOfClass("Humanoid")
    if hum then 
        hum.AutoRotate = true 
    end
    
    State.hittingCooldown = false
end

local KEY_CACHE_FILE = "NovuHub_KeyCache.json"

local function trim(text)
    return tostring(text or ""):gsub("^%s+", ""):gsub("%s+$", "")
end

local function saveCachedKey(key, expiresAt)
    if type(writefile) ~= "function" then return end
    pcall(function()
        writefile(KEY_CACHE_FILE, HttpService:JSONEncode({
            key = key,
            expiresAt = tonumber(expiresAt) or 0
        }))
    end)
end

local function clearCachedKey()
    if type(delfile) == "function" and type(isfile) == "function" then
        pcall(function()
            if isfile(KEY_CACHE_FILE) then delfile(KEY_CACHE_FILE) end
        end)
    elseif type(writefile) == "function" then
        pcall(function() writefile(KEY_CACHE_FILE, "") end)
    end
end

local function loadCachedKey()
    if type(readfile) ~= "function" or type(isfile) ~= "function" then return nil end
    local ok, result = pcall(function()
        if not isfile(KEY_CACHE_FILE) then return nil end
        local raw = readfile(KEY_CACHE_FILE)
        if raw == "" then return nil end
        return HttpService:JSONDecode(raw)
    end)
    if not ok or type(result) ~= "table" then return nil end
    result.key = trim(result.key)
    result.expiresAt = tonumber(result.expiresAt) or 0
    if result.key == "" then return nil end
    return result
end

local function normalizeExpiry(data)
    local nowMs = DateTime.now().UnixTimestampMillis

    -- API expires_at değerini saniye veya milisaniye olarak gönderebilir.
    local expires = tonumber(data.expires_at or data.expiresAt)
    if expires then
        if expires < 100000000000 then
            expires = expires * 1000
        end
        return expires
    end

    -- Bazı API sürümleri yalnızca kalan süreyi saniye olarak gönderir.
    local remaining = tonumber(data.remaining_time or data.remainingTime)
    if remaining and remaining > 0 then
        return nowMs + (remaining * 1000)
    end

    -- Sunucudan süre gelmezse doğrulanan key bu cihazda 24 saat saklanır.
    return nowMs + (24 * 60 * 60 * 1000)
end

local function requestKeyStatus(key)
    key = trim(key)

    local requestFunction =
        request or
        http_request or
        (syn and syn.request) or
        (http and http.request)

    -- Önce yeni Worker API biçimini dene.
    if requestFunction then
        local ok, response = pcall(function()
            return requestFunction({
                Url = KEY_API .. "/api/roblox-verify",
                Method = "POST",
                Headers = {
                    ["Content-Type"] = "application/json",
                    ["Accept"] = "application/json"
                },
                Body = HttpService:JSONEncode({
                    key = key
                })
            })
        end)

        if ok and response then
            local body = response.Body or response.body or ""
            local decodedOk, data = pcall(function()
                return HttpService:JSONDecode(body)
            end)

            if decodedOk and type(data) == "table" then
                if data.success == true or data.valid == true then
                    return true, normalizeExpiry(data), nil
                end

                local reason = tostring(
                    data.error or
                    data.message or
                    data.reason or
                    "Key geçersiz."
                )

                -- Endpoint gerçekten bulunduysa eski API'ye düşmeden hatayı göster.
                local status = tonumber(response.StatusCode or response.Status)
                if status and status ~= 404 and status ~= 405 then
                    return false, normalizeExpiry(data), reason
                end
            end
        end
    end

    -- Eski Worker sürümleriyle de uyumlu kal.
    local ok, response = pcall(function()
        return game:HttpGet(
            KEY_API .. "/verify?key=" .. HttpService:UrlEncode(key)
        )
    end)

    if not ok then
        return false, nil, "API bağlantı hatası."
    end

    local decodedOk, data = pcall(function()
        return HttpService:JSONDecode(response)
    end)

    if not decodedOk or type(data) ~= "table" then
        return false, nil, "Geçersiz API yanıtı."
    end

    if data.valid == true or data.success == true then
        return true, normalizeExpiry(data), nil
    end

    return false, normalizeExpiry(data),
        tostring(data.error or data.message or data.reason or "Key geçersiz.")
end

    clearCachedKey()
end

local function setStatus(text, mode)
    Status.Text = text

    if mode == "success" then
        Status.TextColor3 = Color3.fromRGB(158, 255, 188)
        StatusDot.BackgroundColor3 = Color3.fromRGB(79, 230, 129)

local function statusText(key, extra)
    local value = LANG[currentLanguage][key] or key
    if extra ~= nil then
        return value .. tostring(extra)
    end
    return value
end

local function setTranslatedStatus(key, mode, extra)
    lastStatusKey = key
    lastStatusMode = mode or "idle"
    lastStatusExtra = extra
    setStatus(statusText(key, extra), mode)
end

-- Mobil ve PC sürükleme
do
    local dragging = false
    local dragStart
    local startPos
    local dragInput

    Header.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
            or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = Box.Position

            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    Header.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement
            or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and input == dragInput then
            local delta = input.Position - dragStart
            Box.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
            Shadow.Position = UDim2.new(
                Box.Position.X.Scale,
                Box.Position.X.Offset,
                Box.Position.Y.Scale,
                Box.Position.Y.Offset + 4
            )
        end
    end)
end

local function verifyKey()
    if busy then return end

    local key = trim(KeyBox.Text)
    if key == "" then
        setTranslatedStatus("pasteFirst", "error")
        TweenService:Create(InputWrap, TweenInfo.new(0.06), {
            Position = UDim2.new(0, -5, 0, 96)
        }):Play()
        task.wait(0.06)
        TweenService:Create(InputWrap, TweenInfo.new(0.08), {
            Position = UDim2.new(0, 5, 0, 96)
        }):Play()
        task.wait(0.08)
        TweenService:Create(InputWrap, TweenInfo.new(0.08), {
            Position = UDim2.new(0, 0, 0, 96)
        }):Play()
        return
    end

    busy = true
    VerifyButton.Text = LANG[currentLanguage].verifying
    VerifyButton.Active = false
    setTranslatedStatus("connecting", "loading")

    local valid, expiresAt, reason = requestKeyStatus(key)

    if valid then
        saveCachedKey(key, expiresAt)
        VerifyButton.Text = LANG[currentLanguage].verified
        setTranslatedStatus("opening", "success")

        TweenService:Create(BoxScale, TweenInfo.new(0.28, Enum.EasingStyle.Back, Enum.EasingDirection.In), {
            Scale = 0.88
        }):Play()
        TweenService:Create(Box, TweenInfo.new(0.28), {
            BackgroundTransparency = 1
        }):Play()

        task.wait(0.32)
        KeyGui:Destroy()
        startNovuHub()
        return
    end

    setTranslatedStatus("invalid", "error", tostring(reason or LANG[currentLanguage].unknown))
    VerifyButton.Text = LANG[currentLanguage].retry
    VerifyButton.Active = true
    busy = false
end

local UserInputService = game:GetService("UserInputService")
local function makeDraggable(handle, target)
    local dragging = false
    local dragStart
    local startPosition
    local dragInput

    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPosition = target.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then dragging = false end
            end)
        end
    end)

    handle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and input == dragInput then
            local delta = input.Position - dragStart
            target.Position = UDim2.new(startPosition.X.Scale, startPosition.X.Offset + delta.X, startPosition.Y.Scale, startPosition.Y.Offset + delta.Y)
        end
    end)
end

local function novuSerialize(value, depth)
    depth = depth or 0
    if depth > 6 then return "nil" end

    local valueType = typeof(value)
    if valueType == "boolean" or valueType == "number" then
        return tostring(value)
    elseif valueType == "string" then
        return string.format("%q", value)
    elseif valueType == "table" then
        local parts = {"{"}
        for key, item in pairs(value) do
            local encodedKey
            if type(key) == "string" and key:match("^[%a_][%w_]*$") then
                encodedKey = key
            else
                encodedKey = "[" .. novuSerialize(key, depth + 1) .. "]"
            end
            table.insert(parts, encodedKey .. "=" .. novuSerialize(item, depth + 1) .. ",")
        end
        table.insert(parts, "}")
        return table.concat(parts)
    end
    return "nil"
end

local function novuSaveSettings()
    -- getgenv yedeği, dosya desteği olmayan executorlerde de aynı oturumda çalışır.
    getgenv().NovuHubSavedSettings = novuSaved

    if type(writefile) ~= "function" then
        return false
    end

    local okEncode, payload = pcall(function()
        return HttpService:JSONEncode(novuSaved)
    end)
    if not okEncode then return false end

    local saved = false
    local okRoot = pcall(function()
        writefile(NOVU_SAVE_FILE, payload)
    end)
    saved = okRoot or saved

    if type(makefolder) == "function" then
        pcall(function()
            if type(isfolder) ~= "function" or not isfolder("NovuHub") then
                makefolder("NovuHub")
            end
        end)
        local okFolder = pcall(function()
            writefile(NOVU_SAVE_FILE_FOLDER, payload)
        end)
        saved = okFolder or saved
    end

    return saved
end

local function novuApplyLoadedTable(loaded)
    if type(loaded) ~= "table" then return false end
    for key, value in pairs(loaded) do
        if key == "Keys" and type(value) == "table" then
            novuSaved.Keys = novuSaved.Keys or {}
            for keyName, keyValue in pairs(value) do
                novuSaved.Keys[keyName] = keyValue
            end
        else
            novuSaved[key] = value
        end
    end
    return true
end

local function novuLoadJsonFile(path)
    if type(isfile) ~= "function" or type(readfile) ~= "function" then return false end
    local exists = false
    pcall(function() exists = isfile(path) end)
    if not exists then return false end

    local ok, loaded = pcall(function()
        return HttpService:JSONDecode(readfile(path))
    end)
    return ok and novuApplyLoadedTable(loaded)
end

local function novuLoadLegacyFile(path)
    if type(isfile) ~= "function" or type(readfile) ~= "function" then return false end
    local exists = false
    pcall(function() exists = isfile(path) end)
    if not exists then return false end
    local loader = loadstring or load
    if not loader then return false end
    local ok, loaded = pcall(function()
        local chunk = loader(readfile(path))
        return chunk and chunk()
    end)
    return ok and novuApplyLoadedTable(loaded)
end

local function novuLoadSettings()
    local loaded = novuLoadJsonFile(NOVU_SAVE_FILE)
    if not loaded then loaded = novuLoadJsonFile(NOVU_SAVE_FILE_FOLDER) end

    if not loaded then loaded = novuLoadLegacyFile("NovuHub/settings_v24.lua") end

    if type(getgenv().NovuHubSavedSettings) == "table" then
        novuApplyLoadedTable(getgenv().NovuHubSavedSettings)
    end
end

        local filtered = box.Text:gsub("[^%d%.]", "")
        local number = tonumber(filtered) or currentValue
        number = math.clamp(number, 1, 500)
        number = math.floor(number * 10 + 0.5) / 10
        currentValue = number
        box.Text = tostring(number)
        getgenv()[valueName] = number
        if valueName == "NovuNormalSpeed" then
            novuSaved.NormalSpeed = number
        elseif valueName == "NovuSpeedWhile" then
            novuSaved.CarrySpeed = number
        end
        novuSaveSettings()
        return currentValue
    end

    box:GetPropertyChangedSignal("Text"):Connect(function()
        local filtered = box.Text:gsub("[^%d%.]", "")
        local firstDot = filtered:find("%.")
        if firstDot then
            filtered = filtered:sub(1, firstDot) .. filtered:sub(firstDot + 1):gsub("%.", "")
        end
        if box.Text ~= filtered then
            box.Text = filtered
            box.CursorPosition = #filtered + 1
        end
    end)

    box.Focused:Connect(function()
        TweenService:Create(stroke, TweenInfo.new(0.14), {
            Color = Color3.fromRGB(205, 105, 255),
            Transparency = 0,

            Color = Color3.fromRGB(118, 47, 170),
            Transparency = 0.38,
            Thickness = 1
        }):Play()
    end)

    return box, function()
        return tonumber(getgenv()[valueName]) or currentValue
    end
end

local antiLagEnabled = false
local removeAccessoriesEnabled = false
local antiLagDescConn = nil
local defaultLighting = nil

local function applyAntiLagDerender(object)
    pcall(function()
        if object:IsA("Accessory") or object:IsA("Hat") then
            object:Destroy()
        elseif object:IsA("BasePart") then
            object.Material = Enum.Material.Plastic
            object.Reflectance = 0
            object.CastShadow = false
        elseif object:IsA("Decal") or object:IsA("Texture") then
            object.Transparency = 1
        elseif object:IsA("ParticleEmitter")
            or object:IsA("Trail")
            or object:IsA("Beam")
            or object:IsA("Fire")
            or object:IsA("Smoke")
            or object:IsA("Sparkles") then
            object.Enabled = false
        end
    end)
end

local function enableAntiLag()
    if antiLagEnabled then return end
    removeAccessoriesEnabled = true
    antiLagEnabled = true

    defaultLighting = defaultLighting or {
        Brightness = Lighting.Brightness,
        ClockTime = Lighting.ClockTime,
        OutdoorAmbient = Lighting.OutdoorAmbient,
        GlobalShadows = Lighting.GlobalShadows,
        FogEnd = Lighting.FogEnd,
        EnvironmentDiffuseScale = Lighting.EnvironmentDiffuseScale,
        EnvironmentSpecularScale = Lighting.EnvironmentSpecularScale,
    }

    Lighting.GlobalShadows = false
    Lighting.FogEnd = 1e10
    Lighting.Brightness = 1
    Lighting.EnvironmentDiffuseScale = 0
    Lighting.EnvironmentSpecularScale = 0

    for _, effect in ipairs(Lighting:GetChildren()) do
        pcall(function()
            if effect:IsA("BlurEffect")
                or effect:IsA("SunRaysEffect")
                or effect:IsA("ColorCorrectionEffect")
                or effect:IsA("BloomEffect")
                or effect:IsA("DepthOfFieldEffect") then
                effect.Enabled = false
            end
        end)
    end

    for _, object in ipairs(workspace:GetDescendants()) do
        applyAntiLagDerender(object)
    end

    if antiLagDescConn then
        antiLagDescConn:Disconnect()
    end
    antiLagDescConn = workspace.DescendantAdded:Connect(function(object)
        if removeAccessoriesEnabled then
            applyAntiLagDerender(object)
        end
    end)
end

    antiLagEnabled = false

    if antiLagDescConn then
        antiLagDescConn:Disconnect()
        antiLagDescConn = nil
    end

        pcall(function()
            Lighting.Brightness = defaultLighting.Brightness
            Lighting.ClockTime = defaultLighting.ClockTime
            Lighting.OutdoorAmbient = defaultLighting.OutdoorAmbient
            Lighting.GlobalShadows = defaultLighting.GlobalShadows

local antiRagdollEnabled = false
local antiRagdollRenderConn = nil
local antiRagdollCharacterConn = nil
local ragdollCache = {}

local function cacheRagdollCharacter(character)
    character = character or player.Character
    if not character then return false end

    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local root = character:FindFirstChild("HumanoidRootPart")
    if not humanoid or not root then return false end

    ragdollCache = {
        character = character,
        humanoid = humanoid,
        root = root,
    }

    local camera = workspace.CurrentCamera
    if camera then
        camera.CameraSubject = humanoid
    end
    return true
end

local function isCurrentlyRagdolled()
    local humanoid = ragdollCache.humanoid
    if not humanoid then return false end

    local state = humanoid:GetState()
    if state == Enum.HumanoidStateType.Physics
        or state == Enum.HumanoidStateType.Ragdoll
        or state == Enum.HumanoidStateType.FallingDown then
        return true
    end

    local endTime = player:GetAttribute("RagdollEndTime")
    if endTime then
        local now = workspace:GetServerTimeNow()
        if (endTime - now) > 0 then
            return true
        end
    end

    return false
end

local function removeRagdollConstraints()
    local character = ragdollCache.character
    if not character then return end

    for _, object in ipairs(character:GetDescendants()) do
        if object:IsA("BallSocketConstraint")
            or (object:IsA("Attachment") and object.Name:find("RagdollAttachment")) then
            pcall(function()
                object:Destroy()
            end)
        end
    end
end

local function forceExitRagdoll()
    local humanoid = ragdollCache.humanoid
    local root = ragdollCache.root
    if not humanoid or not root then return end

    pcall(function()
        player:SetAttribute("RagdollEndTime", workspace:GetServerTimeNow())
    end)

    if humanoid.Health > 0 then
        humanoid:ChangeState(Enum.HumanoidStateType.Running)
    end

    root.Anchored = false
    root.AssemblyLinearVelocity = Vector3.zero
    root.AssemblyAngularVelocity = Vector3.zero
    root.CFrame = CFrame.new(root.Position) * CFrame.Angles(0, math.rad(root.Orientation.Y), 0)

    local camera = workspace.CurrentCamera
    if camera and camera.CameraSubject ~= humanoid then
        camera.CameraSubject = humanoid
    end
end

local function stopAntiRagdoll()
    antiRagdollEnabled = false

    if antiRagdollRenderConn then
        antiRagdollRenderConn:Disconnect()
        antiRagdollRenderConn = nil
    end
    if antiRagdollCharacterConn then
        antiRagdollCharacterConn:Disconnect()
        antiRagdollCharacterConn = nil
    end
    table.clear(ragdollCache)
end

local function startAntiRagdoll()
    stopAntiRagdoll()
    antiRagdollEnabled = true
    cacheRagdollCharacter(player.Character)

    antiRagdollRenderConn = RunService.RenderStepped:Connect(function()
        if not antiRagdollEnabled then return end

        if not ragdollCache.humanoid or not ragdollCache.humanoid.Parent then
            cacheRagdollCharacter(player.Character)
            return
        end

        if isCurrentlyRagdolled() then
            removeRagdollConstraints()
            forceExitRagdoll()
        end
    end)

    antiRagdollCharacterConn = player.CharacterAdded:Connect(function(character)
        task.wait(0.5)
        if antiRagdollEnabled then
            cacheRagdollCharacter(character)
        end
    end)
end

    local nearest, dist = nil, math.huge
    for _, plot in ipairs(plots:GetChildren()) do

        local pods = plot:FindFirstChild("AnimalPodiums")
        if not pods then continue end

                end
            end
        end
    end

local function executeSteal(prompt)
    if isStealing then return end

    if not StealData[prompt] then
        StealData[prompt] = {hold = {}, trigger = {}, ready = true}

        if getconnections then
            for _, c in ipairs(getconnections(prompt.PromptButtonHoldBegan)) do
                if c.Function then
                    table.insert(StealData[prompt].hold, c.Function)
                end
            end

            for _, c in ipairs(getconnections(prompt.Triggered)) do
                if c.Function then
                    table.insert(StealData[prompt].trigger, c.Function)
                end
            end
        end
    end

    local data = StealData[prompt]
    if not data.ready then return end

    data.ready = false
    isStealing = true
    local startTime = tick()

    task.spawn(function()
        for _, f in ipairs(data.hold) do
            pcall(f)
        end

        while tick() - startTime < STEAL_DURATION do

local function fadePanels(alpha, duration)
    for _, object in ipairs(guiObjects) do
        if object.Parent then
            local original = originalTransparency[object]
            local goals = {}
            if original.BackgroundTransparency ~= nil then
                goals.BackgroundTransparency = original.BackgroundTransparency + (1 - original.BackgroundTransparency) * alpha
            end
            if original.TextTransparency ~= nil then
                goals.TextTransparency = original.TextTransparency + (1 - original.TextTransparency) * alpha
            end
            if original.TextStrokeTransparency ~= nil then
                goals.TextStrokeTransparency = original.TextStrokeTransparency + (1 - original.TextStrokeTransparency) * alpha
            end
            if original.ImageTransparency ~= nil then
                goals.ImageTransparency = original.ImageTransparency + (1 - original.ImageTransparency) * alpha
            end
            if next(goals) then
                TweenService:Create(object, TweenInfo.new(duration, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), goals):Play()
            end
        end
    end
end

local function findAnyTool()
    local c = LP.Character
    if c then 
        for _, v in ipairs(c:GetChildren()) do 
            if v:IsA("Tool") then return v end 
        end 
    end
    local bp = LP:FindFirstChildOfClass("Backpack")
    if bp then 
        for _, v in ipairs(bp:GetChildren()) do 
            if v:IsA("Tool") then return v end 
        end 
    end
    return nil
end

local function findBat()
    local char = LP.Character
    if not char then return nil end
    
    -- Check equipped tools
    for _, tool in ipairs(char:GetChildren()) do
        if tool:IsA("Tool") then
            local name = tool.Name:lower()
            if name:find("bat") or name:find("slap") then 
                return tool 
            end
        end
    end
    
    -- Check backpack
    local bp = LP:FindFirstChild("Backpack")
    if bp then
        for _, tool in ipairs(bp:GetChildren()) do
            if tool:IsA("Tool") then
                local name = tool.Name:lower()
                if name:find("bat") or name:find("slap") then 
                    return tool 
                end
            end
        end
    end
    
    return nil
end

local function getClosestTarget()
    local root = LP.Character and LP.Character:FindFirstChild("HumanoidRootPart")
    if not root then return nil end
    
    local closest, minDist = nil, math.huge
    
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= LP and plr.Character then
            local tRoot = plr.Character:FindFirstChild("HumanoidRootPart")
            local hum = plr.Character:FindFirstChildOfClass("Humanoid")
            
            if tRoot and hum and hum.Health > 0 then
                local dist = (tRoot.Position - root.Position).Magnitude
                if dist < minDist then 
                    minDist = dist
                    closest = tRoot 
                end
            end
        end
    end
    
    return closest
end

local function tryHitBat()
    if State.hittingCooldown then return end
    State.hittingCooldown = true
    
    pcall(function()
        local c = LP.Character
        if not c then return end
        
        local hum = c:FindFirstChildOfClass("Humanoid")
        local tool = findAnyTool()
        
        if tool then
            -- Equip if not already equipped
            if tool.Parent ~= c and hum then 
                pcall(function() hum:EquipTool(tool) end) 
            end
            
            -- Try to swing
            local remote = tool:FindFirstChildOfClass("RemoteEvent")
            if remote then 
                pcall(function() remote:FireServer() end)
            else 
                pcall(function() tool:Activate() end) 
            end
        end
    end)
    
    task.delay(MOB_SWING_COOLDOWN, function() 
        State.hittingCooldown = false 
    end)
end

            end
        end

        -- Get closest target
        local target = getClosestTarget()
        if not target then return end

        -- Target prediction
        local targetVel = target.AssemblyLinearVelocity
        local myPos = root.Position
        local targetPos = target.Position

        -- Predict position (0.14 seconds ahead + look vector offset)
        local predictPos = targetPos + targetVel * 0.14
        predictPos = predictPos + target.CFrame.LookVector * 0.3

        -- Calculate movement direction
        local direction = predictPos - myPos
        local flatDir = Vector3.new(direction.X, 0, direction.Z).Unit
        local chaseSpeed = 58

        -- Vertical movement with height advantage
        local desiredHeight = targetPos.Y + 3.7
        local yVel = (desiredHeight - myPos.Y) * 19.5 + targetVel.Y * 0.8
        
        if hum.FloorMaterial ~= Enum.Material.Air then
            yVel = math.max(yVel, 13)
        end
        yVel = math.clamp(yVel, -70, 110)

        -- Apply velocity with smoothing
        local desiredVel = Vector3.new(
            flatDir.X * chaseSpeed, 
            yVel, 
            flatDir.Z * chaseSpeed
        )
        root.AssemblyLinearVelocity = root.AssemblyLinearVelocity:Lerp(desiredVel, 0.8)

        -- 3D Look at target with tilt (Dark Hub style)
        local speed3 = targetVel.Magnitude
        local predictTime = math.clamp(speed3 / 150, 0.05, 0.2)
        local predictedPos = targetPos + targetVel * predictTime
        local toPredict = predictedPos - myPos
        
        if toPredict.Magnitude > 0.1 then
            local goalCF = CFrame.lookAt(myPos, predictedPos)
            local curCF = root.CFrame
            local diffCF = curCF:Inverse() * goalCF
            local rx, ry, rz = diffCF:ToEulerAnglesXYZ()
            
            -- Clamp rotations to prevent spinning
            rx = math.clamp(rx, -2.5, 2.5)
            ry = math.clamp(ry, -2.5, 2.5)
            rz = math.clamp(rz, -2.5, 2.5)
            
            local tiltSpeed = 42
            root.AssemblyAngularVelocity = root.CFrame:VectorToWorldSpace(
                Vector3.new(rx * tiltSpeed, ry * tiltSpeed, rz * tiltSpeed)
            )
        end
        
        -- Auto-swing at target (optional)
        -- Uncomment to enable automatic hitting
        -- if (target.Position - myPos).Magnitude < 10 then
        --     tryHitBat()
        -- end
    end)
end

    local c = LP.Character

end

-- çalıştırıldığında key hâlâ geçerliyse key ekranı açılmaz.
local KEY_CACHE_FILE = "NovuHub_KeyCache.json"

local function trim(text)
    return tostring(text or ""):gsub("^%s+", ""):gsub("%s+$", "")
end

local function saveCachedKey(key, expiresAt)
    if type(writefile) ~= "function" then return end
    pcall(function()
        writefile(KEY_CACHE_FILE, HttpService:JSONEncode({
            key = key,
            expiresAt = tonumber(expiresAt) or 0
        }))
    end)
end

local function clearCachedKey()
    if type(delfile) == "function" and type(isfile) == "function" then
        pcall(function()
            if isfile(KEY_CACHE_FILE) then delfile(KEY_CACHE_FILE) end
        end)
    elseif type(writefile) == "function" then
        pcall(function() writefile(KEY_CACHE_FILE, "") end)
    end
end

local function loadCachedKey()
    if type(readfile) ~= "function" or type(isfile) ~= "function" then return nil end
    local ok, result = pcall(function()
        if not isfile(KEY_CACHE_FILE) then return nil end
        local raw = readfile(KEY_CACHE_FILE)
        if raw == "" then return nil end
        return HttpService:JSONDecode(raw)
    end)
    if not ok or type(result) ~= "table" then return nil end
    result.key = trim(result.key)
    result.expiresAt = tonumber(result.expiresAt) or 0
    if result.key == "" then return nil end
    return result
end

local function normalizeExpiry(data)
    local nowMs = DateTime.now().UnixTimestampMillis

    -- API expires_at değerini saniye veya milisaniye olarak gönderebilir.
    local expires = tonumber(data.expires_at or data.expiresAt)
    if expires then
        if expires < 100000000000 then
            expires = expires * 1000
        end
        return expires
    end

    -- Bazı API sürümleri yalnızca kalan süreyi saniye olarak gönderir.
    local remaining = tonumber(data.remaining_time or data.remainingTime)
    if remaining and remaining > 0 then
        return nowMs + (remaining * 1000)
    end

    -- Sunucudan süre gelmezse doğrulanan key bu cihazda 24 saat saklanır.
    return nowMs + (24 * 60 * 60 * 1000)
end

local function requestKeyStatus(key)
    key = trim(key)

    local requestFunction =
        request or
        http_request or
        (syn and syn.request) or
        (http and http.request)

    -- Önce yeni Worker API biçimini dene.
    if requestFunction then
        local ok, response = pcall(function()
            return requestFunction({
                Url = KEY_API .. "/api/roblox-verify",
                Method = "POST",
                Headers = {
                    ["Content-Type"] = "application/json",
                    ["Accept"] = "application/json"
                },
                Body = HttpService:JSONEncode({
                    key = key
                })
            })
        end)

        if ok and response then
            local body = response.Body or response.body or ""
            local decodedOk, data = pcall(function()
                return HttpService:JSONDecode(body)
            end)

            if decodedOk and type(data) == "table" then
                if data.success == true or data.valid == true then
                    return true, normalizeExpiry(data), nil
                end

                local reason = tostring(
                    data.error or
                    data.message or
                    data.reason or
                    "Key geçersiz."
                )

                -- Endpoint gerçekten bulunduysa eski API'ye düşmeden hatayı göster.
                local status = tonumber(response.StatusCode or response.Status)
                if status and status ~= 404 and status ~= 405 then
                    return false, normalizeExpiry(data), reason
                end
            end
        end
    end

    -- Eski Worker sürümleriyle de uyumlu kal.
    local ok, response = pcall(function()
        return game:HttpGet(
            KEY_API .. "/verify?key=" .. HttpService:UrlEncode(key)
        )
    end)

    if not ok then
        return false, nil, "API bağlantı hatası."
    end

    local decodedOk, data = pcall(function()
        return HttpService:JSONDecode(response)
    end)

    if not decodedOk or type(data) ~= "table" then
        return false, nil, "Geçersiz API yanıtı."
    end

    if data.valid == true or data.success == true then
        return true, normalizeExpiry(data), nil
    end

    return false, normalizeExpiry(data),
        tostring(data.error or data.message or data.reason or "Key geçersiz.")
end

        cached.expiresAt = cached.expiresAt * 1000
    end

    if cached.expiresAt > nowMs then
        startNovuHub()
        return
    end

    clearCachedKey()
end

local function addButtonStyle(button, colorA, colorB)
    button.AutoButtonColor = false
    button.BorderSizePixel = 0
    Instance.new("UICorner", button).CornerRadius = UDim.new(0, 11)

    local gradient = Instance.new("UIGradient")
    gradient.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0, colorA),
        ColorSequenceKeypoint.new(1, colorB)
    })
    gradient.Rotation = 12
    gradient.Parent = button

    local stroke = Instance.new("UIStroke")
    stroke.Color = Color3.fromRGB(222, 178, 255)
    stroke.Transparency = 0.76
    stroke.Thickness = 1
    stroke.Parent = button

    button.MouseEnter:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.15), {
            BackgroundTransparency = 0.08
        }):Play()
        TweenService:Create(stroke, TweenInfo.new(0.15), {
            Transparency = 0.38
        }):Play()
    end)

    button.MouseLeave:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.15), {
            BackgroundTransparency = 0
        }):Play()
        TweenService:Create(stroke, TweenInfo.new(0.15), {
            Transparency = 0.76
        }):Play()
    end)

    button.MouseButton1Down:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.08), {
            Size = UDim2.new(button.Size.X.Scale, button.Size.X.Offset - 4, button.Size.Y.Scale, button.Size.Y.Offset - 2)
        }):Play()
    end)

    button.MouseButton1Up:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.1), {
            Size = UDim2.new(button.Size.X.Scale, button.Size.X.Offset + 4, button.Size.Y.Scale, button.Size.Y.Offset + 2)
        }):Play()
    end)
end

local function copyText(text)
    if setclipboard then
        setclipboard(text)
        return true
    elseif toclipboard then
        toclipboard(text)
        return true
    end
    return false
end

-- Mobil ve PC sürükleme
do
    local dragging = false
    local dragStart
    local startPos
    local dragInput

    Header.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
            or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = Box.Position

            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    Header.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement
            or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and input == dragInput then
            local delta = input.Position - dragStart
            Box.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
            Shadow.Position = UDim2.new(
                Box.Position.X.Scale,
                Box.Position.X.Offset,
                Box.Position.Y.Scale,
                Box.Position.Y.Offset + 4
            )
        end
    end)
end

local busy = false

local function verifyKey()
    if busy then return end

    local key = trim(KeyBox.Text)
    if key == "" then
        setStatus("Önce keyini kutuya yapıştır.", "error")
        TweenService:Create(InputWrap, TweenInfo.new(0.06), {
            Position = UDim2.new(0, -5, 0, 96)
        }):Play()
        task.wait(0.06)
        TweenService:Create(InputWrap, TweenInfo.new(0.08), {
            Position = UDim2.new(0, 5, 0, 96)
        }):Play()
        task.wait(0.08)
        TweenService:Create(InputWrap, TweenInfo.new(0.08), {
            Position = UDim2.new(0, 0, 0, 96)
        }):Play()
        return
    end

    busy = true
    VerifyButton.Text = "DOĞRULANIYOR..."
    VerifyButton.Active = false
    setStatus("Sunucuyla bağlantı kuruluyor...", "loading")

    local valid, expiresAt, reason = requestKeyStatus(key)

    if valid then
        saveCachedKey(key, expiresAt)
        VerifyButton.Text = "KEY DOĞRULANDI  ✓"
        setStatus("Key doğru! Novu Hub açılıyor...", "success")

        TweenService:Create(BoxScale, TweenInfo.new(0.28, Enum.EasingStyle.Back, Enum.EasingDirection.In), {
            Scale = 0.88
        }):Play()
        TweenService:Create(Box, TweenInfo.new(0.28), {
            BackgroundTransparency = 1
        }):Play()

        task.wait(0.32)
        KeyGui:Destroy()
        startNovuHub()
        return
    end

    setStatus("Key geçersiz veya süresi dolmuş: " .. tostring(reason or "Bilinmeyen hata"), "error")
    VerifyButton.Text = "TEKRAR DENE  →"
    VerifyButton.Active = true
    busy = false
end

VerifyButton.MouseButton1Click:Connect(verifyKey)

KeyBox.FocusLost:Connect(function(enterPressed)
    if enterPressed then
        verifyKey()
    end
end)
